Here is a complete HTML + CSS + JavaScript code for a Birthday Wish Animation that includes:

✅ Full HTML Code:

html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Birthday Wish</title>
  <style>
    body {
      margin: 0;
      padding: 0;
      background: linear-gradient(#87ceeb, #f0f8ff);
      overflow: hidden;
      font-family: 'Arial', sans-serif;
      text-align: center;
    }

    h1 {
      margin-top: 20px;
      font-size: 3em;
      color: #ff1493;
    }

    .balloon {
      position: absolute;
      width: 40px;
      height: 60px;
      background: red;
      border-radius: 50% 50% 45% 45%;
      animation: float 5s infinite ease-in-out;
      z-index: 1;
    }

    .string {
      position: absolute;
      top: 60px;
      left: 50%;
      width: 2px;
      height: 40px;
      background: black;
    }

    @keyframes float {
      0% { transform: translateY(0); }
      50% { transform: translateY(-100px); }
      100% { transform: translateY(0); }
    }

    .btn {
      margin: 20px;
      padding: 15px 30px;
      font-size: 18px;
      border: none;
      border-radius: 10px;
      cursor: pointer;
      background-color: #ff69b4;
      color: white;
      transition: 0.3s;
    }

    .btn:hover {
      background-color: #ff1493;
    }

    .decoration {
      display: none;
      margin-top: 20px;
      font-size: 2em;
      color: orange;
    }

    .cake {
      display: none;
      margin: 50px auto;
      width: 200px;
      height: 150px;
      background: #d2691e;
      border-radius: 10px;
      position: relative;
    }

    .cake::before {
      content: "";
      position: absolute;
      top: -30px;
      left: 85px;
      width: 30px;
      height: 30px;
      background: white;
      border-radius: 50%;
    }

    .candle {
      position: absolute;
      top: -60px;
      left: 95px;
      width: 10px;
      height: 30px;
      background: #fff;
      display: none;
    }

    .flame {
      width: 10px;
      height: 20px;
      background: orange;
      border-radius: 50%;
      position: absolute;
      top: -20px;
      left: 0;
      animation: flicker 0.2s infinite alternate;
    }

    @keyframes flicker {
      0% { transform: scaleY(1); opacity: 1; }
      100% { transform: scaleY(1.3); opacity: 0.8; }
    }
  </style>
</head>
<body>

  <h1>🎉 Happy Birthday! 🎂</h1>

  <button class="btn" onclick="decorate()">Decorate</button>
  <button class="btn" onclick="bringCake()">Bring Cake</button>

  <div id="decor" class="decoration">🎊🎈Let’s Celebrate!🎈🎊</div>

  <div class="cake" id="cake">
    <div class="candle" id="candle">
      <div class="flame"></div>
    </div>
  </div>

  <script>
    // Create balloons
    for (let i = 0; i < 20; i++) {
      let b = document.createElement("div");
      b.className = "balloon";
      b.style.left = Math.random() * window.innerWidth + "px";
      b.style.backgroundColor = `hsl(${Math.random() * 360}, 100%, 70%)`;
      let s = document.createElement("div");
      s.className = "string";
      b.appendChild(s);
      document.body.appendChild(b);
    }

    function decorate() {
      document.getElementById("decor").style.display = "block";
    }

    function bringCake() {
      const cake = document.getElementById("cake");
      const candle = document.getElementById("candle");
      cake.style.display = "block";
      setTimeout(() => {
        candle.style.display = "block";
      }, 1000);
    }
  </script>

</body>
</html>

🧁 Features:

  • 🎈 20 colorful balloons float automatically.

  • 🎊 "Decorate" button reveals celebration text.

  • 🎂 "Bring Cake" button shows cake and lights candle with a flame.
